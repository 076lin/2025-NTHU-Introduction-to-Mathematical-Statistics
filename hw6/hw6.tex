% 1. 不等式與極限定理

\sectiontitle{1. Inequalities \& Limit Theorems}

\distname{Markov's Inequality:}
\[ P(u(X) \ge c) \le \frac{E[u(X)]}{c} \]

\distname{Chebyshev's Inequality:}
\[ P(|X-\mu| \ge k\sigma) \le \frac{1}{k^2} \]
Also implies: $P(|X-\mu| < k\sigma) \ge 1 - \frac{1}{k^2}$

\distname{Jensen's Inequality:}
\begin{itemize}[leftmargin=*]
    \item If $f''(x) \ge 0$ (convex): $f(E[X]) \le E[f(X)]$
    \item If $f''(x) > 0$ (strictly convex): $f(E[X]) < E[f(X)]$
\end{itemize}

\distname{Weak Law of Large Numbers (WLLN):}
If $X_n$ are i.i.d, mean $\mu$, var $\sigma^2 < \infty$:
\[ \overline{X}_n \xrightarrow{P} \mu \quad \text{as } n \to \infty \]
Proof uses Chebyshev: $P(|\overline{X}_n - \mu| \ge \epsilon) \le \frac{\sigma^2}{n\epsilon^2} \to 0$.

\distname{Strong Law of Large Numbers (SLLN):}
Kolmogorov SLLN: $\overline{X}_n \xrightarrow{a.s.} \mu$ iff $E[X_i] = \mu < \infty$.
\[ P(\omega : \lim_{n \to \infty} X_n(\omega) = \mu) = 1 \]

\distname{Central Limit Theorem (CLT):}
\[ \frac{\overline{X}_n - \mu}{\sigma/\sqrt{n}} \xrightarrow{d} N(0,1) \quad \text{as } n \to \infty \]

\distname{Convergence Types:}
$\xrightarrow{a.s.} \implies \xrightarrow{P} \implies \xrightarrow{d}$.

\sectiontitle{2. Transformation \& Expectation}

\distname{Moment Generating Function (MGF):}
\[ M_X(t) = E[e^{tX}], \quad M^{(m)}(0) = E[X^m] \]
\textbf{Uniqueness:} $F_X(z) = F_Y(z) \iff M_X(t) = M_Y(t)$.
\textbf{Independence:} $f(x_1, x_2) = f(x_1)f(x_2) \iff M(t_1, t_2) = M(t_1)M(t_2)$.
Marginal MGF: $M_{X_1}(t_1) = M_{X_1, X_2}(t_1, 0)$.

\distname{Transformation $Y=g(X)$:}
\[ f_Y(y) = f_X(g^{-1}(y)) \left| \frac{d}{dy}g^{-1}(y) \right| \]
\begin{itemize}[leftmargin=*]
    \item $g$ increasing: $F_Y(y) = F_X(g^{-1}(y))$
    \item $g$ decreasing: $F_Y(y) = 1 - F_X(g^{-1}(y))$
\end{itemize}

\distname{Double Expectation (Adam \& Eve's Law):}
\[ E[X_2] = E[E[X_2|X_1]] \]
\[ Var(X_2) = E[Var(X_2|X_1)] + Var(E[X_2|X_1]) \]

\distname{Covariance:}
\[ Cov(X,Y) = E[XY] - \mu_X \mu_Y \]
\[ Var(aX + bY) = a^2\sigma_X^2 + b^2\sigma_Y^2 + 2ab Cov(X,Y) \]
If $X, Y$ independent $\implies E[XY] = E[X]E[Y]$.

\distname{Bivariate Normal Conditional:}
\[ E[Y|X] = \mu_Y + \rho \frac{\sigma_Y}{\sigma_X}(X - \mu_X) \]
\[ E[Var(Y|X)] = \sigma_Y^2(1-\rho^2) \]

% 3. 離散分佈 (Discrete)
% =========================================
\sectiontitle{3. Discrete Distributions}

\distname{Bernoulli($p$):}
$E[X] = p, \quad Var(X) = p(1-p)$

\distname{Binomial($n, p$):}
$P(X=k) = \binom{n}{k} p^k q^{n-k}$
\\ $E[X] = np, \quad Var(X) = npq$
\\ $M_X(t) = (pe^t + q)^n$

\distname{Geometric($p$):} (Trials until 1st success)
$P(X=x) = p(1-p)^{x-1}$
\\ $E[X] = \frac{1}{p}, \quad Var(X) = \frac{1-p}{p^2}$

\distname{Negative Binomial($r, p$):} ($r$-th success)
$P(Y=y) = \binom{y+r-1}{r-1} p^r (1-p)^y$ (y failures)
\\ $M_Y(t) = \left( \frac{p}{1-(1-p)e^t} \right)^r$

\distname{Poisson($\lambda$):}
$P(X=k) = \frac{\lambda^k e^{-\lambda}}{k!}$
\\ $E[X] = \lambda, \quad Var(X) = \lambda$
\\ $M_X(t) = \exp(\lambda(e^t - 1))$

\distname{Hypergeometric($N, K, n$):}
$P(X=k) = \frac{\binom{K}{k}\binom{N-K}{n-k}}{\binom{N}{n}}$
\\ $E[X] = \frac{nK}{N}, \quad Var(X) = \frac{nK}{N}(1-\frac{K}{N})\frac{N-n}{N-1}$

\distname{Multinomial($n, p_1, \dots, p_k$):}
$P(X_1=x_1, \dots) = \frac{n!}{x_1! \dots x_k!} p_1^{x_1} \dots p_k^{x_k}$
\\ $E[X_i] = np_i, \quad Var(X_i) = np_i(1-p_i)$
\\ $Cov(X_i, X_j) = -np_i p_j$
\\ $M(\mathbf{t}) = (p_1 e^{t_1} + \dots + p_k e^{t_k})^n$

% =========================================
% 4. 連續分佈 (Continuous)
% =========================================
\sectiontitle{4. Continuous Distributions}

\distname{Exponential($\lambda$):}
$f(x) = \lambda e^{-\lambda x}, \quad x \ge 0$
\\ $F(x) = 1 - e^{-\lambda x}$
\\ $E[X] = \frac{1}{\lambda}, \quad Var(X) = \frac{1}{\lambda^2}$

\distname{Gamma($\alpha, \beta$):}
$f(x) = \frac{1}{\Gamma(\alpha)\beta^\alpha} x^{\alpha-1} e^{-x/\beta}$
\\ $E[X] = \alpha\beta, \quad Var(X) = \alpha\beta^2$
\\ $M_X(t) = (1-\beta t)^{-\alpha}, \quad t < 1/\beta$

\distname{Chi-Squared($k$):} ($\chi^2_k \equiv \text{Gamma}(k/2, 2)$)
$E[X] = k, \quad Var(X) = 2k$
\\ $M_X(t) = (1-2t)^{-k/2}$

\distname{Beta($\alpha, \beta$):}
$f(x) = \frac{\Gamma(\alpha+\beta)}{\Gamma(\alpha)\Gamma(\beta)} x^{\alpha-1}(1-x)^{\beta-1}$
\\ $E[X] = \frac{\alpha}{\alpha+\beta}$
\\ $Var(X) = \frac{\alpha\beta}{(\alpha+\beta)^2 (\alpha+\beta+1)}$

\distname{Normal($\mu, \sigma^2$):}
$f(x) = \frac{1}{\sqrt{2\pi}\sigma} e^{-\frac{(x-\mu)^2}{2\sigma^2}}$
\\ $E[X] = \mu, \quad Var(X) = \sigma^2$
\\ $M_X(t) = e^{\mu t + \frac{1}{2}\sigma^2 t^2}$

% =========================================
% 5. 估計與順序統計量
% =========================================
\sectiontitle{5. Estimation \& Order Statistics}

\distname{Maximum Likelihood Estimator (MLE):}
\[ L(\theta) = \prod_{i=1}^n f(x_i; \theta) \]
Maximize $L(\theta)$ or $\ln L(\theta)$ w.r.t $\theta$.

\distname{Order Statistics ($Y_1 < \dots < Y_n$):}
Joint PDF:
\[ g(y_1, \dots, y_n) = n! \prod_{i=1}^n f(y_i) \]
Marginal PDF of $Y_k$:
\[ g_k(y_k) = \frac{n!}{(k-1)!(n-k)!} [F(y_k)]^{k-1} [1-F(y_k)]^{n-k} f(y_k) \]
